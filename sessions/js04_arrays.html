                        let temperaturas = [12, 35, 8, 23, 19, 9, 30, -6];
                        // temperatura mínima, máxima e média

                        let max = -Infinity;
                        let min = +Infinity;
                        let total = 0;
                        for (let temp of temperaturas) {
                            if (temp > max) {
                                max = temp;
                            }
                            if (temp < min) {
                                min = temp;
                            }
                            total += temp;
                        }
                        console.log("Max=" + max);
                        console.log("Min=" + min);
                        let media = total / temperaturas.length;
                        console.log("Média=" + media);
