                        function extraiExtensao(nomeFicheiro) {
                            let dotPosition = nomeFicheiro.indexOf(".");
                            return nomeFicheiro.slice(dotPosition + 1);
                        }
            
                        function validaExtensao(nomeFiheiro) {
                            let validExtensions = ["png", "jpeg", "gif"];
                            let extensao = extraiExtensao(nomeFiheiro);
            
                            // if (validExtensions.indexOf(extensao) >= 0) {
                            //     return true;
                            // } else {
                            //     return false;
                            // }
            
                            // return validExtensions.indexOf(extensao) >= 0 ? true : false;
                            return validExtensions.includes(extensao) ? true : false;
                        }
            
                        console.log(validaExtensao("teste.gif")); // true
