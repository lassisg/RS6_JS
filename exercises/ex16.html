Exercício 16

Crie uma função, chamada separar(), que simule o método split(" ") - isto é, uma função que crie um array a partir de uma string, em que cada elemento do array corresponde a uma palavra da string.

Partindo da string,

                    
                        let texto="Exemplo de um texto que necessita de ser separado";                
                        let resultado=separar(texto);

deve ser produzido um array "resultado" com o seguinte conteúdo: 

                    
                        ["Exemplo", "de", "um", "texto", "que", "necessita", "de","ser", "separado"];

Solução

                    
                        function separar(text) {
                            let pos;
                            let posFinal = text.lastIndexOf(" ");
            
                            let newArray = [];
            
                            do {
                                pos = text.indexOf(" ");
            
                                if (pos === -1) {
                                    newArray.push(text.slice(0));
                                } else {
                                    newArray.push(text.slice(0, pos));
                                }
            
                                text = text.slice(pos + 1);
                            } while (pos !== -1);
            
                            return newArray;
                        }
            
                        let texto = "Exemplo de um texto que necessita de ser separado";
                        let resultado = separar(texto);
                        console.log(texto);
                        console.log(resultado);
