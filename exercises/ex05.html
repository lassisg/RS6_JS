Exercício 5

Simular o método trim() que remove os espaços que ocorrem no início e no fim de uma string.

Exemplo:

                    
                        let frase=" Frase com espaços no início e no fim  ";

Deveremos ter como resultado "Frase com espaços no início e no fim".

Solução

                    
                        let frase = " Frase com espaços no início e no fim   ";

                        // Minha solução (menos performática por fazer substr a cada iteração)
                        while (frase.charAt(0) === " ") {
                            frase = frase.substr(1);
                        }

                        while (frase.charAt(frase.length - 1) === " ") {
                            frase = frase.substr(0, frase.length - 1);
                        }

                        console.log(`"${frase}"`);

                        // Solução 2
                        frase = " Frase com espaços no início e no fim   ";
                        let posInicial = 0;
                        while (frase[posInicial] === " ") {
                            posInicial++;
                        }

                        let posFinal = frase.length - 1;
                        while (frase[posFinal] === " ") {
                            posFinal--;
                        }
                        let fraseFinal = frase.substr(posInicial, posFinal);

                        console.log(posInicial);
                        console.log(posFinal);
                        console.log(`"${fraseFinal}"`);
